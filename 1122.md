class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        vector<int> res;
        unordered_map<int, int> num_cnt;
        set<int> nums;
        for (int num : arr1) {
            ++num_cnt[num];
            nums.insert(num);
        }
        for (int num : arr2) {
            nums.erase(num);
            while (num_cnt[num]--) res.push_back(num);
        }
        for (int num : nums) {
            while (num_cnt[num]--) res.push_back(num);
        }
        return res;
    }
};
