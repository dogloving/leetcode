```cpp
class Solution {
public:
    uint32_t reverseBits(uint32_t n){
        vector<uint16_t> v;
        while(n > 0){
            uint16_t mod = n % 2;
            v.push_back(mod);
            n /= 2;
        }
        while(v.size() < 32){
            v.push_back(0);
        }
        for(auto i : v)cout << i ;
        cout << endl;
        uint32_t result = 0;
        uint32_t pow2 = 1;
        for(int i = v.size()-1; i >= 0;--i){
            if(v[i])result += pow2;
            pow2 *= 2;
        }
    return result;
}
};
```

题意：翻转32位的数字。如00000010100101000001111010011100翻转后为00111001011110000010100101000000。
