var depthSumInverse = function(nestedList) {
    let res = []
    for (let nest of nestedList) helper(nest,0)
    let r = 0
    for (let i = 0; i < res.length; i++) r += res[i]*(res.length-i)
    return r

    function helper(nest, level) {
        if (res.length<level+1) res[level] = 0
        if (nest.isInteger()) res[level] += nest.getInteger()
        else for (let child of nest.getList()) helper(child, level+1)
    }
};
