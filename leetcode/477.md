class Solution {
private:
    void decToBin(int n,vector<int>& count0,vector<int>& count1){
        int count = 0;
        while(n){
            int rem = n % 2;
            if(rem)++count1[count];
            else ++count0[count];
            ++count;
            n /= 2;
        }
        while(count < 100){
            ++count0[count++];
        }
    }
public:
    int totalHammingDistance(vector<int>& nums) {
        int result =0;
        vector<int> count0(100,0),count1(100,0);
        for(int n:nums){
            decToBin(n,count0,count1);
        }
        for(int i = 0;i < 100;++i){
            result += count0[i]*count1[i];
        }
        return result;
    }
};
