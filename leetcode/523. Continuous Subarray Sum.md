```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {boolean}
 */
var checkSubarraySum = function(nums, k) {
    const n = nums.length
    let sum = [0]
    for (let i = 0; i < n; i++) {
        sum[i+1] = sum[i]+nums[i]
    }
    let mod_idx = {}
    for (let i = 1; i <= n; i++) {
        let m = sum[i]%k
        if (m==0&&i>=2) return true
        if ((m in mod_idx)&&(i-mod_idx[m]>=2)) return true
        if (!(m in mod_idx)) mod_idx[m] = i
    }
    return false
};
```

思路：如果nums[i:j]能整除k，则sum[j]%k==sum[i-1]%k。
