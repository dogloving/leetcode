class Solution {
public:
    string reverseVowels(string s) {
        string a="";
        for(int i=0;i<s.length();i++)
            if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U')
                a+=s[i];
        string b="";
        for(int i=a.length()-1;i>=0;i--)
            b+=a[i];
        string ret="";
        int count=0;
        for(int i=0;i<s.length();i++){
            if(!(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U'))
                ret+=s[i];
            else
                ret+=b[count++];
        }
        return ret;
    }
};
