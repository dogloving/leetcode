```CPP
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int inf = 9999999;
        vector<int> dp(amount+1, inf);
        dp[0] = 0;
        for (int i = 1; i <= amount; ++i) {
            for (int coin: coins) {
                if (i-coin>=0) dp[i] = min(dp[i], dp[i-coin]+1);
            }
        }
        return dp[amount]==inf?-1:dp[amount];
    }
};
```
