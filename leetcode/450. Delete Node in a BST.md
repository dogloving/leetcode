```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* deleteNode(TreeNode* root, int key) {
        if(!root)return root;
        //look for
        TreeNode *p=root,*pp=NULL;
        while(p && p->val != key){
            pp=p;
            if(p->val > key)p=p->left;
            else p=p->right;
        }
        
        //not found
        if(!p)return root;
        //found
        //two child
        if(p->left && p->right){
            //if(root->val == key)
            TreeNode* current=p->right,*cc=p;
            while(current->left){
                cc=current;
                current=current->left;
            }
            
            p->val=current->val;
            if(current->right && cc!=p)cc->left=current->right;
            if(current->right == NULL && cc!=p) cc->left=NULL;
            if(current->right && cc ==p)cc->right=current->right;
            if(current->right==NULL && cc==p)cc->right=NULL;
            //delete current;
            return root;
        }
        //one child
        else if(p->left || p->right){
            if(p->left){
                if(root->val == key){
                    root=root->left;
                    return root;
                }
                if(pp->left == p){
                    pp->left=p->left;
                }
                else if(pp->right == p){
                    pp->right=p->left;
                }
              
            }
            else if(p->right){
                if(root->val == key){
                    root=root->right;
                    return root;
                }
                if(pp->left == p){
                    pp->left=p->right;
                }
                else if(pp->right == p){
                    pp->right=p->right;
                }
               
            }
        }
        //no child
        else{
            if(root->val == key){
                root=NULL;
                return root;
            }
            if(pp->left == p){
                pp->left=NULL;
            }
            else if(pp->right == p){
                pp->right=NULL;
            }
         
        }
        return root;
    }
};
```

题意：删除一棵BST中值为key的结点。

思路：删除节点后将左子树最右或右子树最左结点值放到原root。
