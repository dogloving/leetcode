class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        vector<int> res;
        vector<int> l2r, r2l;
        int x = 0;
        for (int n: arr) {
            x ^= n;
            l2r.push_back(x);
        }
        x = 0;
        for (int i = arr.size()-1; i >= 0; --i) {
            x ^= arr[i];
            r2l.push_back(x);
        }
        reverse(r2l.begin(), r2l.end());
        int base = r2l[0];
        cout << base << endl;
        for (vector<int> q: queries) {
            int l = q[0], r = q[1];
            int rr = base;
            if (l > 0) rr ^= l2r[l-1];
            if (r < arr.size()-1) rr ^= r2l[r+1];
            res.push_back(rr);
        }
        return res;
    }
};
