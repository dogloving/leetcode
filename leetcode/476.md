class Solution {
public:
    int findComplement(int num) {
        if(num == 0)return 1;
        bool negative = false;
        if(num < 0){
            negative = true;
            num = -num;
        }
        vector<int> raw;
        while(num > 0){
            raw.push_back(num % 2);
            num /= 2;
        }
        int ret = 0;
        int two = 1;
        for(int i = 0;i < raw.size();++i){
            if(raw[i] == 0){
                ret += two;
            }
            two *= 2;
        }
        return ret;
    }
};
