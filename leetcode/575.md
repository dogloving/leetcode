```
class Solution {
public:
    int distributeCandies(vector<int>& candies) {
        int count = 0;
        set<int> types;
        for(int candy:candies){
            if(types.find(candy) == types.end()){
                ++count;
                types.insert(candy);
            }
            if(count == candies.size() / 2)break;
        }
        return count;
    }
};
```
<pre>其实是贪心的思想。我每次都优先将不同类型的candy放入，所以用set存储。</pre>
