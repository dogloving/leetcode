class Solution {
public:
    int calculate(string s) {
        unordered_map<char, int> op_n;
        op_n['+'] = -1;
        op_n['-'] = -2;
        op_n['*'] = -3;
        op_n['/'] = -4;
        vector<int> exp;
        stack<char> stk_op;
        int n = 0;
        for (int i = 0; i < s.size(); ++i) {
            char ch = s[i];
            if (ch == ' ') continue;
            else if (ch >= '0' && ch <= '9') {
                n = n * 10 + (ch - '0');
                if (i == s.size() - 1 || !(s[i + 1] >= '0' && s[i + 1] <= '9')) {
                    exp.push_back(n);
                    n = 0;
                }
            }
            else if (stk_op.empty()) stk_op.push(ch);
            else if (ch == '+' || ch == '-') {
                while (stk_op.size()) {
                    char op = stk_op.top();
                    if (op != '(') {
                        exp.push_back(op_n[op]);
                        stk_op.pop();
                    }
                    else break;
                }
                stk_op.push(ch);
            }
            else if (ch == '*' || ch == '/') {
                while (stk_op.size()) {
                    char op = stk_op.top();
                    if (op == '*' || op == '/') {
                        exp.push_back(op_n[op]);
                        stk_op.pop();
                    }
                    else break;
                }
                stk_op.push(ch);
            }
            else if (ch == '(') {
                stk_op.push(ch);
            }
            else if (ch == ')') {
                while (stk_op.size()) {
                    char op = stk_op.top();
                    stk_op.pop();
                    if (op == '(') break;
                    exp.push_back(op_n[op]);
                }
            }
        }
        while (stk_op.size()) {
            exp.push_back(op_n[stk_op.top()]);
            stk_op.pop();
        }
        stack<int> stk;
        if (exp.size() == 1) return exp[0];
        for (int n : exp) {
            if (n >= 0) stk.push(n);
            else {
                int a = stk.top();
                stk.pop();
                int b = stk.top();
                stk.pop();
                if (n == -1) stk.push(b + a);
                else if (n == -2) stk.push(b - a);
                else if (n == -3) stk.push(b * a);
                else if (n == -4) stk.push(b / a);
            }
        }
        return stk.top();
    }
};
