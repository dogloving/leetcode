class Solution {
public:
    int longestOnes(vector<int>& A, int K) {
        int start = 0, end = 0;
        int cnt0 = 0;
        int res = 0;
        for (; end < A.size(); ++end) {
            if (A[end] == 0) ++cnt0;
            while (cnt0 > K) {
                if (A[start] == 0) {
                    ++start;
                    --cnt0;
                    break;
                }
                ++start;
            }
            res = max(res, end-start+1);
        }
        return res;
    }
};
