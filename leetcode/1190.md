class Solution {
public:
    string reverseParentheses(string s) {
        while (true) {
            bool found = false;
            for (int i = 0; i < s.size(); ++i) {
                if (s[i] == '(') {
                    found = true;
                    break;
                }
            }
            if (!found) return s;
            string r, tmp;
            int lastIsL = -1;
            for (int i = 0; i < s.size(); ++i) {
                if (lastIsL == -1) {
                    if (s[i] == '(') {
                        lastIsL = 1;
                    } else if (s[i] == ')') {
                        continue;
                    } else r += s[i];
                } else if (lastIsL == 0) {
                    r += s[i];
                } else {
                    if (s[i] == '(') {
                        r += '(';
                        r += tmp;
                        tmp.clear();
                    } else if (s[i] == ')') {
                        reverse(tmp.begin(), tmp.end());
                        r += tmp;
                        tmp.clear();
                        lastIsL = 0;
                    } else tmp += s[i];
                }
            }
            s = r;
        }
    }
};
