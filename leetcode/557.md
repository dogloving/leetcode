 ```
 class Solution {
public:
    string reverseWords(string s) {
        string result;
        string tmp;
        for(char ch:s){
            if(ch == ' '){
                reverse(tmp.begin(),tmp.end());
                result += tmp;
                tmp.clear();
                result += ' ';
            }else{
                tmp += ch;
            }
        }
        reverse(tmp.begin(),tmp.end());
        result += tmp;
        return result;
    }
};
 ```
