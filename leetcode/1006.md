class Solution {
public:
    int clumsy(int N) {
        vector<int> factors;
        int n = N;
        while (n >= 1) {
            if (n >= 4) {
                factors.push_back(n*(n-1)/(n-2));
                factors.push_back(n-3);
                n -= 4;
                continue;
            }
            else if (n == 3) factors.push_back(n*(n-1)/(n-2));
            else if (n == 2) factors.push_back(n*(n-1));
            else factors.push_back(n);
            break;
        }
        int res = factors[0];
        for (int i = 1; i < factors.size(); ++i) {
            if (i % 2) res += factors[i];
            else res -= factors[i];
        }
        return res;
    }
};
