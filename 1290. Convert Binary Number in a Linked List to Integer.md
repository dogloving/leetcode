```CPP
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    int getDecimalValue(ListNode* head) {
        int res = 0;
        while(head) {
            res = res*2 + head->val;
            head = head->next;
        }
        return res;
    }
};
```

题意：0/1单向链表表示二进制数，返回其对应十进制数。
