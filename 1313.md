class Solution {
public:
    vector<int> decompressRLElist(vector<int>& nums) {
        vector<int> res;
        for (int i = 0; i < nums.size(); i+=2) {
            int cnt = nums[i], num = nums[i+1];
            while (cnt--) {
                res.push_back(num);
            }
        }
        return res;
    }
};
