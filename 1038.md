/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* bstToGst(TreeNode* root) {
        node = NULL;
        helper(root);
        return root;
    }
private:
    TreeNode* node;
    void helper(TreeNode* root) {
        if (root == NULL) return;
        if (root->right) helper(root->right);
        if (node) root->val += node->val;
        node = root;
        helper(root->left);
    }
};
