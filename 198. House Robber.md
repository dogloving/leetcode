```cpp
class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        if (n==0) return 0;
        vector<int> xuan(n, 0), buxuan(n, 0);
        xuan[0] = nums[0];
        buxuan[0] = 0;
        for (int i = 1; i < n; ++i) {
            xuan[i] = buxuan[i-1]+nums[i];
            buxuan[i] = max(xuan[i-1], buxuan[i-1]);
        }
        return max(xuan[n-1], buxuan[n-1]);
    }
};
```

题意：抢劫一条街上的商户。每个商户有确定的金钱。不能同时抢相邻的两个商户。问最多可以抢到多少钱。
