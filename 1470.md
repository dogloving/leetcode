class Solution {
public:
    vector<int> shuffle(vector<int>& nums, int n) {
        vector<int> l(nums.begin(), nums.begin()+n), r(nums.begin()+n, nums.end());
        vector<int> res;
        for (int i = 0; i < n; ++i) {
            res.push_back(l[i]);
            res.push_back(r[i]);
        }
        return res;
    }
};
