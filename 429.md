```CPP
/*
// Definition for a Node.
class Node {
public:
    int val = NULL;
    vector<Node*> children;

    Node() {}

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
*/
class Solution {
public:
    vector<vector<int>> levelOrder(Node* root) {
        vector<vector<int>> res;
        if (root == NULL) return res;
        queue<Node*> q;
        q.push(root);
        while (q.size()) {
            vector<int> nums;
            int cnt = q.size();
            while (cnt--) {
                Node* front = q.front();
                q.pop();
                nums.push_back(front->val);
                for (Node* node: front->children) {
                    if (node) q.push(node);
                }
            }
            res.push_back(nums);
        }
        return res;
    }
};
```
