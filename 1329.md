```CPP
class Solution {
public:
    vector<vector<int>> diagonalSort(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        for (int row = 0; row < m; ++row) {
            int i = row, j = 0;
            vector<int> tmp;
            while (i < m && j < n) {
                tmp.push_back(mat[i][j]);
                ++i;
                ++j;
            }
            sort(tmp.begin(), tmp.end());
            i = row, j = 0;
            int idx = 0;
            while (i < m && j < n) {
                mat[i][j] = tmp[idx++];
                ++i;
                ++j;
            }
        }
        for (int col = 1; col < n; ++col) {
            int i = 0, j = col;
            vector<int> tmp;
            while (i < m && j < n) {
                tmp.push_back(mat[i][j]);
                ++i;
                ++j;
            }
            sort(tmp.begin(), tmp.end());
            i = 0, j = col;
            int idx = 0;
            while (i < m && j < n) {
                mat[i][j] = tmp[idx++];
                ++i;
                ++j;
            }
        }
        return mat;
    }
};
```

题意：对每一斜列进行排序，而非全部打乱后排序。
