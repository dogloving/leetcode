class Solution {
public:
     void merge(vector<int>& nums1, int m, vector<int>& nums2, int n){
         vector<int>fuck;
         int i=0,j=0;
         while(i!=m && j!=n){
             if(nums1[i] < nums2[j]){
                 fuck.push_back(nums1[i]);
                 i++;
             }
             else{
                 fuck.push_back(nums2[j]);
                 j++;
             }
         }
         while(i!=m)
            fuck.push_back(nums1[i++]);
        while(j!=n)
            fuck.push_back(nums2[j++]);
        for(int i=0;i<m+n;i++)
        nums1[i]=fuck[i];
     };
};
