class Solution {
public:
    int maxPower(string s) {
        if (s.size() == 0) return 0;
        int res = 1;
        char ch = s[0];
        int cnt = 1;
        for (int i = 1; i < s.size(); ++i) {
            char c = s[i];
            if (c == ch) ++cnt;
            else {
                res = max(res, cnt);
                cnt = 1;
                ch = c;
            }
        }
        return max(res, cnt);
    }
};
