```CPP
class Solution {
public:
    vector<int> runningSum(vector<int>& nums) {
        if (nums.size()==0) return {};
        vector<int> res = {nums[0]};
        for (int i = 1; i < nums.size(); ++i)  res.push_back(res.back()+nums[i]);
        return res;
    }
};
```

题意：给数组nums，求sums.其中sums[i]=sum(nums[0],...,nums[i])
