class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int,int> num_cnt;
        for (int num: arr) ++num_cnt[num];
        unordered_set<int> s;
        for (auto it = num_cnt.begin(); it != num_cnt.end(); ++it) {
            if (s.count(it->second)) return false;
            s.insert(it->second);
        }
        return true;
    }
};
