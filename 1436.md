class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> res;
        for (vector<string>& path: paths) {
            res.insert(path[0]);
            res.insert(path[1]);
        }
        for (vector<string>& path: paths) {
            res.erase(path[0]);
        }
        return *(res.begin());
    }
};
