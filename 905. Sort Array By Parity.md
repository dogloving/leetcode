```CPP
// time: O(n), space: O(1)
class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        int l = 0, r = A.size()-1;
        while (l<r) {
            // find even number
            while (l<r&&A[l]%2==0) ++l;
            // find odd number
            while (l<r&&A[r]%2==1) --r;
            if (l>=r) break;
            // swap A[l] and A[r]
            int tmp = A[l];
            A[l] = A[r];
            A[r] = tmp;
            ++l;
            --r;
        }
        return A;
    }
};
```

题意：给一个非负整数数组A，要求进行排序将偶数放在前面，奇数放在后面。

思路：双指针分别向中间靠。遇到左边奇数右边偶数就swap。
