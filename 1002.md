```CPP
class Solution {
public:
    vector<string> commonChars(vector<string>& A) {
        vector<int> v(26, 9999);
        for (string s: A) {
            vector<int> tmp(26, 0);
            for (char ch: s) ++tmp[ch-'a'];
            for (int i = 0; i < 26; ++i) v[i] = min(v[i], tmp[i]);
        }
        vector<string> res;
        for (int i = 0; i < 26; ++i) {
            if (v[i] != 9999) {
                while (v[i]--) {
                    string s;
                    s.push_back('a'+i);
                    res.push_back(s);
                }
            }
        }
        return res;
    }
};
```

题意：找出每个单词都包含的字母。
