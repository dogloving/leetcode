class Solution {
public:
    vector<int> createTargetArray(vector<int>& nums, vector<int>& index) {
        int n = nums.size();
        vector<int> target(nums.size());
        for (int i = 0; i < n; ++i) {
            for (int j = n-2; j >= index[i]; --j) {
                target[j+1] = target[j];
            }
            target[index[i]] = nums[i];
        }
        return target;
    }
};
