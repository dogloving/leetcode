class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<int> r(n, 0), c(m, 0);
        for (vector<int> indice: indices) {
            ++r[indice[0]];
            ++c[indice[1]];
        }
        int res = 0;
        for (int rr = 0; rr < n; ++rr) {
            for (int cc = 0; cc < m; ++cc) {
                res += (r[rr]+c[cc])&1;
            }
        }
        return res;
    }
};
