class Solution {
public:
    string freqAlphabets(string s) {
        string res;
        for (int i = s.size()-1; i >= 0;) {
            if (s[i] >= '0' && s[i] <='9') {
                res.push_back('a'+(s[i]-'1'));
                --i;
            } else {
                int n = (s[i-2]-'0')*10+(s[i-1]-'0');
                res.push_back('a'+n-1);
                i -= 3;
            }
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
