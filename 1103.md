class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> res(num_people, 0);
        int idx = 0, next = 1;
        while (candies) {
            if (candies >= next) {
                res[idx] += next;
                candies -= next;
            } else {
                res[idx] += candies;
                candies = 0;
            }
            idx = (idx + 1) % num_people;
            ++next;
        }
        return res;
    }
};
