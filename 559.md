```CPP
/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> children;

    Node() {}

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
*/
class Solution {
public:
    int maxDepth(Node* root) {
        return helper(root);
    }
private:
    int helper(Node* node) {
        if (node == NULL) return 0;
        int h = 0;
        for (Node* child: node->children) {
            h = max(h, helper(child));
        }
        return h + 1;
    }
};
````
