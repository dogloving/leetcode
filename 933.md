class RecentCounter {
public:
    RecentCounter() {
        
    }
    
    int ping(int t) {
        ts.push_back(t);
        int cnt = 0;
        for (int i = ts.size() - 1; i >= 0; --i) {
            if (ts[i] >= t - 3000) ++cnt;
            else break;
        }
        return cnt;
    }
private:
    vector<int> ts;
};

/**
 * Your RecentCounter object will be instantiated and called as such:
 * RecentCounter* obj = new RecentCounter();
 * int param_1 = obj->ping(t);
 */
