#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
int count0(string s) {
	int res = 0;
	for (char ch : s) {
		if (ch == '0') ++res;
		else break;
	}
	return res;
}
bool helper(string& s1, string& s2) {
	int z1 = count0(s1), z2 = count0(s2);
	if (z1 != z2) return z1 > z2;
	string temp1 = s1 + s2, temp2 = s2 + s1;
	for (int i = 0; i < temp1.size(); ++i) {
		if (temp1[i] != temp2[i]) return temp1[i] < temp2[i];
	}
	return true;
}
int main()
{
	int n;
	scanf("%d", &n);
	vector<string> ss;
	for (int i = 0; i < n; ++i) {
		string s;
		cin >> s;
		ss.push_back(s);
	}
	sort(ss.begin(), ss.end(), [](string& s1, string& s2) {return helper(s1, s2); });
	bool ok = false;
	for (int i = 0; i < n; ++i) {
		if (!ok) {
			int index = 0;
			while (index < ss[i].size()) {
				if (ss[i][index] != '0') {
					ok = true;

			}

	return 0;
}

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	scanf("%d", &n);
	vector<string> nums(n);
	for (int i = 0; i < n; ++i) cin >> nums[i];
	sort(nums.begin(), nums.end(), [](string n1, string& n2)->bool{
		if (n1 == n2) return true;
		for (int i = 0; i < min(n1.size(), n2.size()); ++i) if (n1[i] != n2[i]) return n1[i] < n2[i];
		if (n1 + n2 > n2 + n1) return false;
		return true;
	});
	string res;
	for (int i = 0; i < n; ++i) res += nums[i];
	for ( int i = 0; i < res.size(); ++i) {
		if (res[i] != '0') {
			res = string(res.begin() + i, res.end());
			break;
		}
	}
	if (res[0] == '0') cout << 0 << endl;
	else cout << res << endl;
}
