```CPP
class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& A) {
        vector<bool> res;
        int n = 0;
        for (int i = 0; i < A.size(); ++i) {
            n = n * 2 + A[i];
            if (n % 5 == 0) res.push_back(true);
            else res.push_back(false);
            n %= 5;
        }
        return res;
    }
};
```
