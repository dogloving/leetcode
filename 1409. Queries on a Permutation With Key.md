```CPP
class Solution {
struct Node {
    int val;
    Node *next;
    Node(int v) {
        val = v;
        next = NULL;
    }
};
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        int n = queries.size();
        vector<int> res;
        Node *head = helper(m);
        for (int i = 0; i < n; ++i) {
            int target = queries[i];
            Node *node = head;
            if (node->val == target) {
                res.push_back(0);
                continue;
            }
            int idx = 1;
            while (node->next) {
                if (node->next->val == target) {
                    res.push_back(idx);
                    Node* next = node->next->next;
                    node->next->next = head;
                    head = node->next;
                    node->next = next;
                    break;
                } else node = node->next;
                ++idx;
            }
        }
        return res;
    }
private:
    Node* helper(int m) {
        Node *head = NULL, *pre = NULL;
        for (int i = 1; i <= m; ++i) {
            Node* node = new Node(i);
            if (head == NULL) head = node;
            else pre->next = node;
            pre = node;
        }
        return head;
    }
};
```

题意：正整数数组queries，值在1~m，执行以下操作。

一开始P=[1,2,3,...,m]；

对当前i，找到queries[i]在P中的位置，并将其移到P最前面。

返回每次queries[i]在P中的位置。
