class Solution {
public:
    vector<int> sortedSquares(vector<int>& A) {
        vector<int> res;
        int index = 0;
        for (int i = 1; i < A.size(); ++i) {
            if (abs(A[i]) < abs(A[index])) index = i;
        }
        res.push_back(pow(A[index], 2));
        int l = index - 1, r = index + 1;
        while (l >= 0 && r < A.size()) {
            if (abs(A[l]) <= abs(A[r])) {
                res.push_back(pow(A[l], 2));
                --l;
            } else {
                res.push_back(pow(A[r], 2));
                ++r;
            }
        }
        while (l >= 0) {
            res.push_back(pow(A[l], 2));
            --l;
        }
        while (r < A.size()) {
            res.push_back(pow(A[r], 2));
            ++r;
        }
        return res;
    }
};
