```CPP
class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> even, odd;
        for (int num: A) {
            if (num & 1) odd.push_back(num);
            else even.push_back(num);
        }
        for (int num: odd) even.push_back(num);
        return even;
    }
};
```
