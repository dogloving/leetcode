```CPP
/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> children;

    Node() {}

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
*/
class Solution {
public:
    vector<int> preorder(Node* root) {
        vector<int> res;
        helper(root, res);
        return res;
    }
private:
    void helper(Node* node, vector<int>& res) {
        if (node == NULL) return;
        res.push_back(node->val);
        for (Node* child: node->children) {
            helper(child, res);
        }
    }
};
```
